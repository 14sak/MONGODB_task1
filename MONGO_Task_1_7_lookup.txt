SQL QUERY:

select c.customer_name,s.ship_mode from customer as c
inner join sales as s
on s.customer_id=c.customer_id
group by c.customer_name,s.ship_mode


MONGODB:

db.customer.aggregate({$group:{_id:{Cust_name:"$customer_name",Ship_mode:"$ship_mode"},sum:{$sum:1}}},{
$lookup:{
from:"sales",
localField:"_id.customer_id",
foreignField:"customer_id",
as:"Joins in Mongo"
}},{$match:{"Joins in Mongo":{ $exists: true, $ne: [] }}}).pretty()
